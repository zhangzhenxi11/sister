### 三、关键数据结构

```typescript
// 1. 学习资料 (Material)
interface Material {
  id: string;
  title: string;
  description: string;
  type: 'video' | 'audio' | 'image' | 'pdf' | 'text';
  filePath: string;
  status: 'processing' | 'completed';
  summary?: string;      // 摘要
  content?: string;      // 完整内容
  createdAt: Date;
}

// 2. 人格 (Persona)
interface Persona {
  id: string;
  name: string;           // "小恋", "情感专家"
  description: string;    // 人格描述
  style: 'gentle' | 'rational' | 'humorous' | 'custom';
  systemPrompt: string;   // 详细的系统提示词
  knowledgeGraph: KnowledgeNode[];  // 知识图谱
  rules: PersonaRule[];   // 行为规则
  stats: PersonaStats;    // 统计信息
  createdAt: Date;
  updatedAt: Date;
}

// 3. 知识图谱节点 (KnowledgeNode)
interface KnowledgeNode {
  id: string;
  type: 'concept' | 'strategy' | 'example' | 'rule';
  content: string;        // 知识内容
  sourceMaterialId?: string;  // 来源资料
  weight: number;         // 权重
  createdAt: Date;
}

// 4. 处理上下文
interface ProcessingContext {
  userMessage: string;
  conversationHistory: Array<{ role: string; content: string }>;
  useRag: boolean;
}
```
